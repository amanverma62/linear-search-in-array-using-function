#include <iostream>

using namespace std;

bool search(int arr[], int n, int key){

    for(int i=0; i<=n; i++){

        if(arr[i]==key){

            return 1;
        }

    }
     return 0;



}


int main()
{

    int key;
    cin>>key;
    int arr[4]= {22,55,1,8};

    bool found = search(arr ,4, key );

    if(found){

            cout<<"key is present"<<endl;
    }
    else{
        cout<<"key is not present"<<endl;
    }






   return 0;

}
